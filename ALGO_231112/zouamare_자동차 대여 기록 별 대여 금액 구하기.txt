SELECT A.HISTORY_ID, (A.END_DATE - A.START_DATE + 1) * B.DAILY_FEE * (1 - ((CASE
    WHEN (A.END_DATE - A.START_DATE + 1) BETWEEN 7 AND 29 THEN (
        SELECT TO_NUMBER(REGEXP_REPLACE(C.DISCOUNT_RATE, '[%]', ''))
        FROM CAR_RENTAL_COMPANY_DISCOUNT_PLAN C
        WHERE B.CAR_TYPE = C.CAR_TYPE
        AND C.DURATION_TYPE = '7일 이상'
    )
    WHEN (A.END_DATE - A.START_DATE + 1) BETWEEN 30 AND 89 THEN (
        SELECT TO_NUMBER(REGEXP_REPLACE(C.DISCOUNT_RATE, '[%]', ''))
        FROM CAR_RENTAL_COMPANY_DISCOUNT_PLAN C
        WHERE B.CAR_TYPE = C.CAR_TYPE
        AND C.DURATION_TYPE = '30일 이상'
    )
    WHEN (A.END_DATE - A.START_DATE + 1) >= 90 THEN (
        SELECT TO_NUMBER(REGEXP_REPLACE(C.DISCOUNT_RATE, '[%]', ''))
        FROM CAR_RENTAL_COMPANY_DISCOUNT_PLAN C
        WHERE B.CAR_TYPE = C.CAR_TYPE
        AND C.DURATION_TYPE = '90일 이상'
    )
    ELSE 0
END) / 100)) AS "FEE"
FROM CAR_RENTAL_COMPANY_RENTAL_HISTORY A,
CAR_RENTAL_COMPANY_CAR B
WHERE A.CAR_ID = B.CAR_ID
AND B.CAR_TYPE = '트럭'
ORDER BY FEE DESC, HISTORY_ID DESC;
