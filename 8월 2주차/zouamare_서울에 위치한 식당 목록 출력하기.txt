SELECT A.REST_ID, A.REST_NAME, A.FOOD_TYPE, A.FAVORITES, A.ADDRESS, B.SCORE
FROM REST_INFO A,
(
    SELECT REST_ID, ROUND(AVG(REVIEW_SCORE), 2) AS SCORE
    FROM REST_REVIEW
    GROUP BY REST_ID
) B
WHERE 
A.REST_ID = B.REST_ID
AND substr(A.address, 1, 2) = '서울'
ORDER BY B.SCORE DESC, A.FAVORITES DESC;
